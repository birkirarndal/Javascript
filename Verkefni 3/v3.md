## 4
```
class Flaug {
    constructor(name) {
    this.name = name;
    this.speed = 10;
    this.life = 10;
    }
    fly() {
        this.speed++;
    }
    get name() {
        return this._name;
    }

    set name(value) {
        this._name = value;
    }

    get speed() {
        return this._speed;
    }

    set speed(value) {
        this._speed = value;
    }

    get life() {
        return this._life;
    }

    set life(value) {
        this._life = value;
    }
}

class Lifeflaug extends Flaug {
    constructor(name, speed, life) {
        super(name, speed, life);
    }
    setLife() {
        this.life++;
    }
}
let f1 = new Lifeflaug("Starfall");
let f2 = new Flaug("Inferno");
let f3 = new Flaug("Neutron");
```
